for feature in geojson_data["features"]:

    province_info = smoking_control_data.get(province_name, {
        "level": 1,
        "level_name": "1级（薄弱）",
        "smoking_rate": "未公开",
        "policy": "暂无数据"
    })
    feature["properties"].update(province_info)
